function show(id,btn){

  document.querySelectorAll('.view').forEach(v=>v.classList.remove('on'));

  document.querySelectorAll('.nav-btn').forEach(b=>b.classList.remove('on'));

  document.getElementById(id).classList.add('on');

  btn.classList.add('on');

}


function toggleDay(id){

  document.getElementById('body-'+id).classList.toggle('open');

}


function markDay(id,status){

  const d=document.getElementById('dbtn-'+id);

  const s=document.getElementById('sbtn-'+id);

  const dot=document.getElementById('dot-'+id);

  if(status==='done'){d.classList.add('active');s.classList.remove('active');dot.style.boxShadow='0 0 0 2px #4caf50';}

  else{s.classList.add('active');d.classList.remove('active');dot.style.boxShadow='0 0 0 2px #f44336';}

  localStorage.setItem('day-'+id,status);

}


function toggleCheck(el){

  el.classList.toggle('chk');

  el.textContent=el.classList.contains('chk')?'✓':'';

}


function buildWeek(){

  const grid=document.getElementById('week-grid');

  DAYS.forEach(d=>{

    const saved=localStorage.getItem('day-'+d.id);

    const isDone=saved==='done'||d.done;

    const isSkip=saved==='skip';

    const card=document.createElement('div');

    card.className='day-card'+(d.today?' today':'');

    card.innerHTML=`

      <div class="day-hdr" onclick="toggleDay('${d.id}')">

        <div class="dot ${d.dot}" id="dot-${d.id}" style="${isDone?'box-shadow:0 0 0 2px #4caf50':isSkip?'box-shadow:0 0 0 2px #f44336':''}"></div>

        <div class="day-nm">${d.name}${d.today?' <span style="font-size:10px;color:#f4621f">· Aujourd\'hui</span>':''}</div>

        <div class="day-dt">${d.date}</div>

        <span class="day-badge ${d.badge}">${d.label}</span>

        <span style="font-size:13px">${isDone?'✅':isSkip?'⏭':'▸'}</span>

      </div>

      <div class="day-body${d.today?' open':''}" id="body-${d.id}">

        <div class="day-desc">${d.desc}</div>

        ${d.today&&d.type==='renfort'?`<button class="action-btn" onclick="document.querySelectorAll('.nav-btn')[1].click()">Voir la séance détaillée →</button>`:''}

          <button id="dbtn-${d.id}" class="btn-done${isDone?' active':''}" onclick="markDay('${d.id}','done')">✓ Fait</button>

}
